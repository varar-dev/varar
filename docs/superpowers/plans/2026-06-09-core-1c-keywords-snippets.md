`packages/bdd/scripts/sync-keywords.ts`:
```ts
// Fetches cucumber/gherkin's gherkin-languages.json from the official repo
// and emits packages/bdd/src/keywords-data.ts. Run on demand, not in CI.

import { writeFileSync } from 'node:fs'
import { resolve } from 'node:path'

const URL =
  'https://raw.githubusercontent.com/cucumber/gherkin/main/gherkin-languages.json'

async function main() {
  const res = await fetch(URL)
  if (!res.ok) throw new Error(`fetch ${URL}: ${res.status}`)
  const data = (await res.json()) as Record<
    string,
    {
      name: string
      native: string
      given: ReadonlyArray<string>
      when: ReadonlyArray<string>
      then: ReadonlyArray<string>
      and: ReadonlyArray<string>
      but: ReadonlyArray<string>
    }
  >
  // Build a flat array of every keyword (trimmed, '* ' wildcard excluded).
  const set = new Set<string>()
  for (const lang of Object.values(data)) {
    for (const group of ['given', 'when', 'then', 'and', 'but'] as const) {
      for (const kw of lang[group]) {
        const trimmed = kw.trim().replace(/[:\s]+$/, '').trim()
        if (trimmed.length === 0) continue
        if (trimmed === '*') continue
        set.add(trimmed)
      }
    }
  }
  const sorted = [...set].sort()
  const out = `// AUTO-GENERATED by scripts/sync-keywords.ts. Do not edit by hand.\nexport const KEYWORDS: ReadonlyArray<string> = ${JSON.stringify(sorted, null, 2)}\n`
  const target = resolve(import.meta.dirname, '..', 'src', 'keywords-data.ts')
  writeFileSync(target, out)
  console.error(`wrote ${target} (${sorted.length} keywords)`)
}

main().catch((err) => {
  console.error(err)
  process.exit(1)
})
```

```ts
import { KEYWORDS } from './keywords-data.js'

const PREFIXES: ReadonlyArray<string> = [...KEYWORDS].sort((a, b) => b.length - a.length)

export function isKeywordLed(sentence: string): boolean {
  const trimmed = sentence.trimStart()
  for (const kw of PREFIXES) {
    if (startsWithKeyword(trimmed, kw)) return true
  }
  return false
}

export function stripLeadingKeyword(sentence: string): string {
  const trimmed = sentence.trimStart()
  for (const kw of PREFIXES) {
    if (startsWithKeyword(trimmed, kw)) {
      const rest = trimmed.slice(kw.length).replace(/^\s+/, '')
      return rest
    }
  }
  return sentence
}

function startsWithKeyword(sentence: string, kw: string): boolean {
  if (!sentence.toLowerCase().startsWith(kw.toLowerCase())) return false
  const next = sentence.charAt(kw.length)
  if (next === '') return true
  // Keyword must be followed by whitespace or punctuation, not another letter.
  return /\s|[,;:]/.test(next)
}
```

```ts
import { stripLeadingKeyword } from './keywords.js'
import type { Registry } from './registry.js'

export type Snippet = {
  readonly expression: string
  readonly handlerSignature: string
  readonly fullCode: string
}

type Token =
  | { readonly kind: 'int' }
  | { readonly kind: 'float' }
  | { readonly kind: 'string' }

const INT_RE = /\b\d+\b/
const FLOAT_RE = /\b\d+\.\d+\b/
const STRING_RE = /"[^"]*"/

const PARAM_NAMES: Record<Token['kind'], string> = {
  int: 'count',
  float: 'price',
  string: 'name',
}

export function generateSnippet(rawText: string, _registry: Registry): Snippet {
  const text = stripLeadingKeyword(rawText.trim())
  const params: Token[] = []
  let cursor = 0
  let expr = ''

  while (cursor < text.length) {
    const slice = text.slice(cursor)
    const floatMatch = FLOAT_RE.exec(slice)
    const intMatch = INT_RE.exec(slice)
    const stringMatch = STRING_RE.exec(slice)
    const candidates = [
      floatMatch ? { kind: 'float' as const, match: floatMatch } : null,
      intMatch && (!floatMatch || intMatch.index < floatMatch.index)
        ? { kind: 'int' as const, match: intMatch }
        : null,
      stringMatch ? { kind: 'string' as const, match: stringMatch } : null,
    ].filter((c): c is { kind: Token['kind']; match: RegExpExecArray } => c !== null)
    if (candidates.length === 0) {
      expr += slice
      break
    }
    candidates.sort((a, b) => a.match.index - b.match.index)
    const best = candidates[0]
    if (!best) {
      expr += slice
      break
    }
    expr += slice.slice(0, best.match.index)
    expr += `{${best.kind}}`
    params.push({ kind: best.kind })
    cursor += best.match.index + best.match[0].length
  }

  const usedNames = new Map<string, number>()
  const handlerArgs = params.map((p) => {
    const baseName = PARAM_NAMES[p.kind]
    const count = (usedNames.get(baseName) ?? 0) + 1
    usedNames.set(baseName, count)
    const name = count === 1 ? baseName : `${baseName}${count}`
    const type = p.kind === 'string' ? 'string' : 'number'
    return `${name}: ${type}`
  })
  const handlerSignature = `(ctx, ${handlerArgs.join(', ')}) => {`
  const fullCode = `step('${expr}', ${handlerSignature}\n  // ...\n})`

  return { expression: expr, handlerSignature, fullCode }
}
```

NOTE: `int` and `float` both yield `count: number` / `price: number` by default but they map to different cucumber-expression types. Name choice is purely a hint to the engineer — the cucumber expression token `{int}`/`{float}` is the load-bearing piece.

- [ ] **Step 3: Extend `generateSnippet` to scan custom parameter types first**

In `packages/bdd/src/snippet.ts`, before the built-in tokenizer loop iteration, try each registered custom parameter type. Each `ParameterType` exposes its name and regular expressions. Iterate them, pick the earliest match, emit `{name}`.

Update the param shape to carry the chosen `name` and a TS type guess (default `string`):

```ts
type Token =
  | { readonly kind: 'int'; readonly name: 'int'; readonly tsType: 'number' }
  | { readonly kind: 'float'; readonly name: 'float'; readonly tsType: 'number' }
  | { readonly kind: 'string'; readonly name: 'string'; readonly tsType: 'string' }
  | { readonly kind: 'custom'; readonly name: string; readonly tsType: 'string' }

function customMatches(text: string, registry: Registry): { name: string; index: number; length: number } | undefined {
  const params = (registry.parameterTypes as unknown as { parameterTypes: Iterable<{ name: string; regexpStrings: ReadonlyArray<string> }> })
    .parameterTypes
  let best: { name: string; index: number; length: number } | undefined
  for (const p of params) {
    if (p.name === 'int' || p.name === 'float' || p.name === 'string' || p.name === 'word' || p.name === '') continue
    for (const regexpString of p.regexpStrings) {
      const re = new RegExp(regexpString)
      const m = re.exec(text)
      if (!m) continue
      const candidate = { name: p.name, index: m.index, length: m[0].length }
      if (!best || candidate.index < best.index) best = candidate
    }
  }
  return best
}
```

Then in the main scan loop, take `customMatches(slice, registry)` and compare its index against `intMatch`/`floatMatch`/`stringMatch`. Take the earliest. When the winner is `custom`, emit `{name}` and record `tsType: 'string'`.

Update `PARAM_NAMES` and the handler-arg naming logic to use the chosen `name` directly for custom types.

## Plan summary

After Plan 1c, the core has every capability the spec assigns to it for v1:

- Gherkin keyword JSON embedded at build time (i18n out of the box).
- `isKeywordLed` and `stripLeadingKeyword` available for downstream callers (planner uses both).
- Snippet generator that infers `{int}` / `{float}` / `{string}` plus any custom parameter types registered in the `Registry`, and produces both a cucumber expression and a typed handler signature.
- Planner emits `missing-step` diagnostics for keyword-led sentences with no match, including a paste-ready snippet.
- Planner emits `orphan-attachment` warnings for tables and fences that don't qualify for `DataTable`/`DocString` attachment.
- Full public surface ready for Plan 2 (the vitest adapter) to consume.

Plans 1, 1b, and 1c together complete `@oselvar/bdd`'s contribution to v1. Plan 2 onwards is shells (adapters, CLI) on top of this pure core.
