## Examples

The examples below give a taste of how Vár works. If the markdown disagrees with the code, your tests will fail.

<Tabs syncKey="example">
  <TabItem label="Deep Thought">
    <Editor>
      <File uri="file:///deep-thought.md">
        <Version label="Fix it">{deepThoughtSource}</Version>
        <Version label="Break it">{deepThoughtSource.replace('42', '41')}</Version>
      </File>
      <File uri="file:///deep-thought.steps.ts">
        <Version label="Fix it">{deepThoughtSteps}</Version>
        <Version label="Break it">{deepThoughtSteps.replace('42', '43')}</Version>
      </File>
    </Editor>
  </TabItem>
  <TabItem label="Yahtzee">
    <Editor>
      <File uri="file:///yahtzee.md">{yahtzeeSource}</File>
      <File uri="file:///yahtzee.steps.ts">{yahtzeeSteps}</File>
      <File uri="file:///yahtzee.ts">{yahtzeeLogic}</File>
    </Editor>
  </TabItem>
  <TabItem label="Roman numerals">
    <Editor>
      <File uri="file:///roman-numerals.md">
